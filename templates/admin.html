代理网关统一入口: http://127.0.0.1:8888/v1

向客户展示时，所有请求应当直接落入该地址。

import requests

url = "http://127.0.0.1:8888/v1/chat/completions"
headers = {
    "Authorization": "Bearer <这里填写网关签发的专属 API Key>",
    "Content-Type": "application/json"
}
data = {
    "model": "gpt-3.5-turbo",
    "messages": [{"role": "user", "content": "你好，测试连接"}]
}
response = requests.post(url, headers=headers, json=data)
print(response.json())
curl -X POST "http://127.0.0.1:8888/v1/chat/completions" \
     -H "Content-Type: application/json" \
     -H "Authorization: Bearer <这里填写网关签发的专属 API Key>" \
     -d '{
        "model": "gpt-3.5-turbo",
        "messages": [{"role": "user", "content": "你好，测试连接"}]
     }'
