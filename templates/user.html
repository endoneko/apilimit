import requests

url = "http://127.0.0.1:8888/v1/chat/completions"
headers = {
    "Authorization": "Bearer <填入您的 API 密钥>",
    "Content-Type": "application/json"
}
data = {
    "model": "gpt-3.5-turbo",
    "messages": [{"role": "user", "content": "你好！"}]
}
response = requests.post(url, headers=headers, json=data)
print(response.json())
